--Query #1
with
cust_agg (cust, min_q, max_q, avg_q) as
	(select cust, min(quant), max(quant), avg(quant)
	from sales
	group by cust),
min_info (cust, min_prod, min_date, min_state) as
	(select cust_agg.cust, prod, date, state
	from cust_agg, sales
	where cust_agg.cust = sales.cust and cust_agg.min_q = sales.quant),
max_info (cust, max_prod, max_date, max_state) as
	(select cust_agg.cust, prod, date, state
	from cust_agg, sales
	where cust_agg.cust = sales.cust and cust_agg.max_q = sales.quant)
select distinct *
from max_info natural join min_info natural join cust_agg
order by max_info.cust


--Query #2
with
total_quants (year, month, day, sum_q) as
	(select year, month, day, sum(quant)
	from sales
	group by year, month, day),
agg_quants (year, month, busiest_total_q, slowest_total_q) as
	(select year, month, max(sum_q), min(sum_q)
	from total_quants
	group by year, month),
busiest_days (year, month, busiest_day, busiest_total_q) as
	(select agg_quants.year, agg_quants.month, day, busiest_total_q
	from total_quants, agg_quants
	where total_quants.year = agg_quants.year
		and total_quants.month = agg_quants.month
		and total_quants.sum_q = agg_quants.busiest_total_q
	order by year, month),
slowest_days (year, month, slowest_day, slowest_total_q) as
	(select agg_quants.year, agg_quants.month, day, slowest_total_q
	from total_quants, agg_quants
	where total_quants.year = agg_quants.year
		and total_quants.month = agg_quants.month
		and total_quants.sum_q = agg_quants.slowest_total_q
	order by year, month)
select * 
from busiest_days natural join slowest_days


--Query #3
with
prod_count (cust, prod, count_q) as
	(select cust, prod, count(quant)
	from sales
	group by cust, prod),
count_agg (cust, max_count, min_count) as
	(select cust, max(count_q),min(count_q)
	from prod_count
	group by cust),
most_fav_prod (cust, most_fav_prod) as
	(select count_agg.cust, prod
	from prod_count, count_agg
	where prod_count.cust = count_agg.cust
		and prod_count.count_q = count_agg.max_count),
least_fav_prod (cust, least_fav_prod) as
	(select count_agg.cust, prod
	from prod_count, count_agg
	where prod_count.cust = count_agg.cust
		and prod_count.count_q = count_agg.min_count)
select * 
from most_fav_prod natural join least_fav_prod


--Query #4
with
avg_spring (cust, prod, spring_avg) as
	(select cust, prod, avg(quant)
	from sales
	where month in (3, 4, 5)
	group by cust, prod),
avg_summer (cust, prod, summer_avg) as
	(select cust, prod, avg(quant)
	from sales
	where month in (6, 7, 8)
	group by cust, prod),
avg_fall (cust, prod, fall_avg) as
	(select cust, prod, avg(quant)
	from sales
	where month in (9, 10, 11)
	group by cust, prod),
avg_winter (cust, prod, winter_avg) as
	(select cust, prod, avg(quant)
	from sales
	where month in (12, 1, 2)
	group by cust, prod),
year_agg (cust, prod, average, total, count) as
	(select cust, prod, avg(quant), sum(quant), count(quant)
	from sales
	group by cust, prod)
select *
from avg_spring 
	natural join avg_summer
	natural join avg_fall 
	natural join avg_winter
	natural join year_agg
order by cust, prod


--Query 5
with
q1_max (prod, q1_max) as
	(select prod, max(quant)
	from sales
	where month in (1, 2, 3)
	group by prod),
q1_date (prod, q1_max, q1_date) as
	(select q1_max.prod, q1_max, date
	from sales, q1_max
	where sales.prod = q1_max.prod
		and sales.quant = q1_max.q1_max
		and month in (1, 2, 3)),
q2_max (prod, q2_max) as
	(select prod, max(quant)
	from sales
	where month in (4, 5, 6)
	group by prod),
q2_date (prod, q2_max, q2_date) as
	(select q2_max.prod, q2_max, date
	from sales, q2_max
	where sales.prod = q2_max.prod
		and sales.quant = q2_max.q2_max
		and month in (4, 5, 6)),
q3_max (prod, q3_max) as
	(select prod, max(quant)
	from sales
	where month in (7, 8, 9)
	group by prod),
q3_date (prod, q3_max, q3_date) as
	(select q3_max.prod, q3_max, date
	from sales, q3_max
	where sales.prod = q3_max.prod
		and sales.quant = q3_max.q3_max
		and month in (7, 8, 9)),
q4_max (prod, q4_max) as
	(select prod, max(quant)
	from sales
	where month in (10, 11, 12)
	group by prod),
q4_date (prod, q4_max, q4_date) as
	(select q4_max.prod, q4_max, date
	from sales, q4_max
	where sales.prod = q4_max.prod
		and sales.quant = q4_max.q4_max
		and month in (10, 11, 12))
select *
from q1_date natural join q2_date natural join q3_date natural join q4_date
