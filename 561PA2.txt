--QUERY #1
WITH
monthly_avg AS
	(SELECT prod, month, avg(quant) avg_quant
	FROM sales
	GROUP BY prod, month),
prev_month AS
	(SELECT m.prod, m.month, avg(pm.avg_quant) prev_avg
	FROM monthly_avg m LEFT JOIN monthly_avg pm 
	ON m.month - 1 = pm.month
	GROUP BY m.prod, m.month),
next_month AS 
	(SELECT m.prod, m.month, avg(nm.avg_quant) next_avg
	FROM monthly_avg m LEFT JOIN monthly_avg nm 
	ON m.month + 1 = nm.month
	GROUP BY m.prod, m.month),
reference AS
	(SELECT prev_month.prod, prev_month.month, prev_avg, next_avg
	FROM prev_month NATURAL JOIN next_month),
sales_between AS
	(SELECT s.*
	FROM sales s JOIN reference r
	ON s.prod = r.prod AND s.month = r.month
	WHERE s.quant between r.prev_avg AND r.next_avg OR s.quant between r.next_avg AND r.prev_avg)
SELECT r.prod PRODUCT, r.month AS MONTH, count(sb.quant) SALES_COUNT_BETWEEN_AVGS
FROM reference r LEFT JOIN sales_between sb
ON r.prod = sb.prod AND r.month = sb.month
GROUP BY r.prod, r.month
ORDER BY r.prod, r.month

--QUERY #2
WITH
ext_sales AS
	(SELECT cust, prod, month, ceiling(month / 3.0) qtr, quant
	FROM sales),
qtr_avgs AS
	(SELECT cust, prod, qtr, avg(quant)
	FROM ext_sales
	GROUP BY cust, prod, qtr),
before_qtr AS
	(SELECT cqtr.cust, cqtr.prod, cqtr.qtr, cqtr.avg, bqtr.avg BEFORE_AVG
	FROM qtr_avgs cqtr LEFT JOIN qtr_avgs bqtr
	ON cqtr.qtr - 1 = bqtr.qtr AND cqtr.cust = bqtr.cust AND cqtr.prod = bqtr.prod),
after_qtr AS
	(SELECT cqtr.cust, cqtr.prod, cqtr.qtr, aqtr.avg AFTER_AVG
	FROM qtr_avgs cqtr LEFT JOIN qtr_avgs aqtr
	ON cqtr.qtr + 1 = aqtr.qtr AND cqtr.cust = aqtr.cust AND cqtr.prod = aqtr.prod)
SELECT b.cust CUSTOMER, b.prod PRODUCT, b.qtr QRTR, b.BEFORE_AVG, b.avg DURING_AVG, a.AFTER_AVG
FROM before_qtr b FULL OUTER JOIN after_qtr a
ON b.cust = a.cust AND b.prod = a.prod AND b.qtr = a.qtr
ORDER BY b.cust, b.prod, b.qtr

--QUERY #3
WITH
base_avg AS
	(SELECT cust, prod, state, avg(quant)
	FROM SALES
	GROUP BY cust, prod, state),
cust_avg AS
	(SELECT ba.cust, ba.prod, ba.state, ba.avg, avg(s.quant) OTHER_CUST_AVG
	FROM base_avg ba, sales s
	WHERE ba.prod = s.prod AND ba.state = s.state AND ba.cust != s.cust
	GROUP BY ba.cust, ba.prod, ba.state, ba.avg),
prod_avg AS
	(SELECT ba.cust, ba.prod, ba.state, avg(s.quant) OTHER_PROD_AVG
	FROM base_avg ba, sales s
	WHERE ba.cust = s.cust AND ba.state = s.state AND ba.prod != s.prod
	GROUP BY ba.cust, ba.prod, ba.state)
SELECT ca.cust CUSTOMER, ca.prod PRODUCT, ca.state STATE, ca.avg PROD_AVG, ca.OTHER_CUST_AVG, pa.OTHER_PROD_AVG
from cust_avg ca NATURAL JOIN prod_avg pa
ORDER BY ca.cust, ca.prod, ca.state

--QUERY #4
WITH
ordered_pos AS
	(SELECT s1.prod, s1.quant,
		(SELECT count(*)
		FROM sales s2
		WHERE s1.prod = s2.prod AND s2.quant <= s1.quant) POS
	FROM sales s1
	ORDER BY s1.prod, s1.quant),
median_pos AS
	(select prod, ceiling(count(*) / 2.0) MEDIAN_POS
	FROM sales
	GROUP BY prod)
SELECT op.prod, min(op.quant)
FROM ordered_pos op, median_pos mp
WHERE op.prod = mp.prod AND op.POS >= mp.MEDIAN_POS
GROUP BY op.prod
